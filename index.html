/* ls-core/engine.ts */
import { AntiSybil, ProxyManager } from '@ls/core';

async function executeFarm(wallets) {
for (const wallet of wallets) {
const proxy = await ProxyManager.assign(wallet);
const delay = AntiSybil.getRandomDelay(300, 1200);

// Await human-like delay
await sleep(delay);

try {
await wallet.interact({
target: 'zksync',
action: 'random_dapp',
proxy: proxy
});
} catch (error) {
Logger.error('Tx Failed', error);
}
}
}
